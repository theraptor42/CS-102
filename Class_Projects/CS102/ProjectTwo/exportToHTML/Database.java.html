44       /* 
45       Method: parseInputLine - text parser 
46       Purpose: parse the line of text and return an entry object 
47       Parameters: 
48           String line   the line of text being parsed 
49       Returns: 
50           Entry - the Object parsed from the text line 
51       */ 
52       private Entry parseInputLine(String line) 
53       { 
54           if (line.equals("")) 
55           { 
56               //returns a null object if the line is empty 
57               return null; 
58           } 
59    
60           //a new empty Entry object 
61           Entry returnNode = new Entry(); 
62           //Scanner to scan the line of text 
63           Scanner lineScanner = new Scanner(line); 
64           lineScanner.useDelimiter("/"); 
65           //Sets the returnNode's word field 
66           returnNode.setWord(lineScanner.next()); 
67    
68           while (lineScanner.hasNext()) 
69           { 
70               //the next word in the line 
71               String nextWord = lineScanner.next(); 
72               if (!(nextWord == null)) 
73               { 
74                   //adds each word in alphabet order to the 
75                   //synonym linkedList 
76                   returnNode.addSynonymInOrder(nextWord); 
77                   //might not have any synonyms 
78               } 
79           } 
80           //returns the finished entry object 
81           return returnNode; 
82       }

84       /* 
85       Method: addWordInOrder - add to linkedList 
86       Purpose: add an entry to the linked list 
87               in alphabetical order 
88       Parameters: 
89           Entry newEntry  the entry being added 
90       Returns: 
91           boolean - did I add it successfully 
92       */ 
93       public boolean addWordInOrder(Entry newEntry) 
94       { 
95           //getting ready to walk down the chain 
96           //the current entry in the chain 
97           Entry current = head; 
98           //the previous entry in the chain 
99           Entry previous = new Entry(); 
100   
101          //the word in the entry being added 
102          String newWord = newEntry.getWord(); 
103          if (current == null) 
104          { 
105              //adds it if the list is empty 
106              head = newEntry; 
107              return true; 
108          } 
109          while (current != null) 
110          { 
111              //the word in the current entry object on the chain 
112              String currentWord = current.getWord(); 
113              if (newWord.compareToIgnoreCase(currentWord) < 0) 
114              { 
115                  //if the newWord comes before the currentWord 
116                  //insert it before it 
117   
118                  //inserts it before the current word 
119                  newEntry.setNextEntry((current)); 
120                  //if it becomes the first word in the list 
121                  //previous will be empty 
122                  if(previous.getWord() == null) 
123                  { 
124                      //previous isn't null because that made 
125                      //other errors, but the word will be null 
126                      //if its an empty entry 
127                      //if it comes before the first word in 
128                      //the list, insert it at the beginning 
129                      head = newEntry; 
130                  } 
131                  else 
132                  { 
133                      //links the previous entry to finish the chain 
134                      previous.setNextEntry(newEntry); 
135                  } 
136                  //return success 
137                  return true; 
138              } 
139              if (newWord.compareToIgnoreCase(currentWord) == 0) 
140              { 
141                  return false;//synonym already exists 
142              } 
143   
144              //move on to next synonym if neither are true 
145              previous = current; 
146              current = current.getNextEntry(); 
147          } 
148          //if it reaches the end of the list, 
149          //add it to the end of the list 
150          previous.setNextEntry(newEntry); 
151          newEntry.setNextEntry(null); 
152          return true; 
153      }

166      { 
167          if (head == null) 
168              return "\nThere are no entries in the thesaurus"; 
169          //the current entry in the chain 
170          Entry current = head; 
171          while ( current != null) 
172          { 
173              if (current.getWord().equalsIgnoreCase(word))

175                  return current.toString();
