129      /* 
130      Method: addSynonymInOrder - add to linkedList 
131      Purpose: add an synonym to the linked list 
132              in alphabetical order 
133      Parameters: 
134          String newSynonym  the synonym being added 
135      Returns: 
136          boolean - did I add it successfully 
137      */ 
138      public boolean addSynonymInOrder(String newSynonym) 
139      { 
140          if (newSynonym.equals("")) 
141          { 
142              //fails if line the synonym is empty 
143              return false; 
144          } 
145          //the current synonym in the chain 
146          SynNode current = firstSynonymNode; 
147          //theprevious synonym in the chain 
148          SynNode previous = null; 
149          //the new synonym node bing add to the chain 
150          SynNode newSynNode = new SynNode(newSynonym, null); 
151          if (current == null) 
152          { 
153              //adds it if the chain is empty 
154              this.setFirstSynonymNode(newSynNode); 
155              return true;//returns a successful addition 
156          } 
157          while (current != null) 
158          { 
159              //the current synonym on the chain 
160              String currentString = current.getSynonym(); 
161              if (newSynonym.compareToIgnoreCase(currentString) < 0) 
162              { 
163                  //if the new synonym goes before the current synonym 
164                  //insert it before it 
165                  newSynNode.setNextSynonymNode(current); 
166                  if(previous == null) 
167                  { 
168                      this.setFirstSynonymNode(newSynNode); 
169                  } 
170                  else 
171                  { 
172                      previous.setNextSynonymNode(newSynNode); 
173                  } 
174                  return true; 
175              } 
176              if (newSynonym.compareToIgnoreCase(currentString) == 0) 
177              { 
178                  return false;//synonym already exists 
179              } 
180   
181              //move on to next synonym if neither are true 
182              previous = current; 
183              current = current.getNextSynonymNode(); 
184          } 
185          //if it reaches the end of the list, 
186          //add it to the end of the list 
187          previous.setNextSynonymNode(newSynNode); 
188          newSynNode.setNextSynonymNode(null); 
189          return true; 
190      }

193      Method: searchForSynonym - search database 
194      Purpose: search the synonym linked list to see if 
195              it has a particular synonym 
196      Parameters: 
197          String Synonym  the synonym being searched for 
198      Returns: 
199          boolean - did I find it?

201      public boolean searchForSynonym(String synonym) 
202      { 
203          if (firstSynonymNode == null) 
204          { 
205              //if there are no synonym return false 
206              return false; 
207          } 
208          //cuurent synNode on the chain 
209          SynNode current = firstSynonymNode; 
210          while (current != null) 
211          { 
212              if (current.getSynonym().equalsIgnoreCase(synonym)) 
213              { 
214                  //return true if you find it 
215                  return true; 
216              } 
217              current = current.getNextSynonymNode(); 
218          } 
219   
220          //return false if you don't find it 
221          return false; 
222      }

242              return null; 
243          //return false if the chain is empty 
244

246          { 
247              if (current.getSynonym().equalsIgnoreCase(synonym)) 
248              { 
249                  //set return me if I find it and 
250                  //break out of the loop 
251                  returnMe = current; 
252                  break; 
253              } 
254              current = current.getNextSynonymNode(); 
255          } 
256   
257          return returnMe;//return what I found

272          //prints out the word and all of its synonyms

286              { 
287                  returnString += ", " + currentSyn.getSynonym(); 
288                  currentSyn = currentSyn.getNextSynonymNode(); 
289              }
