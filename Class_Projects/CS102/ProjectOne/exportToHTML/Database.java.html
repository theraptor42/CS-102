7    import java.util.*; 
8    public class Database //the thesaurus is the database of entries 
9    { 
10       private Entry [] database; 
11       //database holds an array of entries 
12    
13    
14       /* 
15       Method: Database - constructor 
16       Purpose: Scans the input file and creates an Entry object 
17                   for each line with data 
18       Parameters: 
19           Scanner inputScanner:    scanner of the file being parsed 
20       Returns: 
21           Database           the object created by parsing the file 
22       */ 
23       public Database(Scanner inputScanner) 
24       //takes a scanner so main can handle file exceptions 
25       //also so only main has to deal with i/o 
26       { 
27           database = new Entry [1]; 
28           //gives it size one as an initial value 
29           int counter = 0; 
30           while (inputScanner.hasNextLine()) 
31           { 
32               String nextLine = inputScanner.nextLine(); 
33               if (nextLine.equals("")) 
34               {}//skips empty lines 
35               else 
36               { 
37                   if (counter == database.length) { 
38                       Entry[] bigger = Arrays.asList(database) 
39                               .toArray(new Entry[database.length + 1]); 
40                       //increases the size of database by one 
41                       //so I don't have null elements 
42                       // which will break my toString() 
43                       database = bigger; 
44                   } 
45                   database[counter] = new Entry(nextLine); 
46                   //sends the next scanner line to make a new entry 
47                   counter++; 
48               } 
49           } 
50       } 
51    
52       /* 
53       Method: searchForWord - database search 
54       Purpose: Searches database for matching word 
55                   and returns the results of the search 
56       Parameters: 
57           String word      the word to be found 
58       Returns: 
59           String        What we found/not found 
60       */ 
61       public String searchForWord(String word) 
62       { 
63           for (int i = 0; i < database.length; i++) 
64           { 
65               //runs through all the entry objects in the database 
66               if (database[i].getWord().equalsIgnoreCase(word)) 
67               { 
68                   String wordFound = database[i].toString(); 
69                   return wordFound; 
70                   //if it finds the word, breaks out and returns it 
71               } 
72           } 
73           //else it sends back a failure message 
74           String searchFailed = "Sorry, I could not find \"" + word + "\""; 
75           return searchFailed; 
76       } 
77    
78       /* 
79       Method: searchForSynonym - database search 
80       Purpose: Searches each entry database for matching 
81                   synonym and returns the results of the search 
82       Parameters: 
83           String synonym      the synonym to be found 
84       Returns: 
85           String        What we found/not found 
86       */ 
87       public String searchForSynonym(String synonym) 
88       { 
89           String returnString = ""; 
90           for (int i = 0; i < database.length; i++) 
91           { 
92               //loops through each entry object to scan list of synonyms 
93               String [] synonymList = database[i].getSynonymsList(); 
94               for (int j = 0; j < synonymList.length; j++) 
95               { 
96                   if (synonym.equalsIgnoreCase(synonymList[j])) 
97                   { 
98                       //if it finds an entry with that synonym 
99                       //it adds it to the return string 
100                      returnString += database[i].toString() + "\n"; 
101                  } 
102              } 
103          } 
104          if (returnString.equals("")) 
105          { 
106              //if no synonyms are found, sends back failure message 
107              returnString = "Sorry, I could not find any words " + 
108                      "with a synonym \"" + synonym + "\""; 
109          } 
110   
111          return returnString; 
112      } 
113   
114      /* 
115      Method: toString - toString 
116      Purpose: Returns a formatted string containing the useful 
117                  information contained by this Database object 
118      Parameters: 
119          none 
120      Returns: 
121          String           Formatted string with word and synonyms 
122                              from eah entry object 
123      */ 
124      public String toString() 
125      { 
126          //uses the entry's toString 
127          //loops through each entry and adds it toString to the return 
128          String returnString = ""; 
129          for (int i =0; i < database.length; i++) 
130          { 
131              returnString += database[i].toString()+"\n"; 
132          } 
133          return returnString; 
134      }
