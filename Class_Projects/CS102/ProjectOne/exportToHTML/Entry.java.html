7    import java.util.*; 
8    public class Entry 
9    { 
10       private String word;//contains the word of the entry 
11       private String[] synonyms;//contains the synonyms of the word 
12    
13       /* 
14       Method: Entry - constructor 
15       Purpose: Parse a line of text into a String word and 
16                       String array synonyms 
17       Parameters: 
18           String line:        the line to be parsed 
19       Returns: 
20           Entry               the object created from text line 
21       */ 
22       public Entry(String line) 
23       { 
24           Scanner lineScanner = new Scanner(line); 
25           lineScanner.useDelimiter("/"); 
26    
27           word = lineScanner.next(); 
28           //empty lines are caught in database 
29    
30           synonyms = new String[0]; 
31           int counter = 0; 
32           String nextWord; 
33           while (lineScanner.hasNext()) 
34           { 
35               nextWord = lineScanner.next(); 
36               //gets next word to be evaluated 
37               //for inclusion in the array 
38               if (nextWord.equals("")) 
39               {}//excludes any null elements from my array 
40               else 
41               { 
42                   if (counter == synonyms.length) 
43                   { 
44                       //increases the size of the array by one 
45                       //so array contains no null elements 
46                       String [] bigger = Arrays.asList(synonyms) 
47                               .toArray(new String [synonyms.length+1]); 
48                       synonyms = bigger; 
49                   } 
50                   synonyms[counter] = nextWord; 
51                   //stores the synonym into the array 
52                   counter++; 
53               } 
54           } 
55       } 
56    
57       /* 
58       Method: getWord - getter 
59       Purpose: Returns word for external use 
60       Parameters: 
61           none 
62       Returns: 
63           String               the Entry's word 
64       */ 
65       public String getWord() 
66       { 
67           return word; 
68           //returns word for external use 
69           //because word is private 
70       } 
71    
72       /* 
73       Method: getSynonymList - getter 
74       Purpose: Returns synonyms for external use 
75       Parameters: 
76           none 
77       Returns: 
78           String []           the String array of synonyms 
79       */ 
80       public String [] getSynonymsList() 
81       { 
82           return synonyms; 
83           //returns synonyms for external use 
84           //because synonyms is private 
85       } 
86    
87       /* 
88       Method: toString - toString 
89       Purpose: Returns a formatted string containing the useful 
90                   information contained by this Entry object 
91       Parameters: 
92           none 
93       Returns: 
94           String           Formatted string with word and synonyms 
95       */ 
96       public String toString() 
97       { 
98           String returnString = "Word: " + word + "\nSynonyms: "; 
99           if (synonyms.length > 0) 
100          { 
101              returnString += synonyms[0]; 
102              //loops and gets all the synonyms for this entry 
103              for (int i = 1; i < synonyms.length; i++) { 
104                  returnString += ", " + synonyms[i]; 
105              } 
106          } 
107          else 
108          //if none are found, returns failure message 
109              returnString += "No synonyms found"; 
110          return returnString; 
111          //puts the entry into a string in the format 
112          //Word: word 
113          //Synonyms: syn1, syn2, ... 
114      } 
115  }
